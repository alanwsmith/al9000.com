wasm - hello, world

This is a basic `Hello, World` WASM app written in rust.

Cargo.toml

[package]
name = "hello-world"
version = "0.1.0"
edition = "2024"

[lib]
crate-type = ["cdylib"]

[dependencies]
wasm-bindgen = "0.2.116"

src/lib.rs

use wasm_bindgen::prelude::*;

#[wasm_bindgen]
pub fn hello_world() -> String {
  "Hello, world!".to_string()
}

HTML

<div id="output"></div>

JavaScript

<script type="module">

  import init, {hello_world} from "./pkg/hello_world.js";

  async function main() {
    await init();
    const el = document.querySelector("#output");
    el.innerHTML = hello_world();
  }

  main();

</script>

Build Script

#!/bin/bash

wasm-pack build --target web
