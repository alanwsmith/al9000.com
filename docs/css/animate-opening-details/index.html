Animate Opening a details Element

NOTE: This page doesn't have base styles applied yet so things look kinda busted.

HTML

<details>
  <summary>Details to expand</summary>
  <div class="default-flow">
    <p>
      Lorem ipsum aliquet. Platea velit maximus. 
    </p>
    <p>
      Cras fusce rhoncus ligula bibendum. Nunc varius urna aliquet. 
    </p>
    <p>
      Tempor quis maecenas congue dapibus eros.
    </p>
  </div>
</details>

CSS

:root {
  interpolate-size: allow-keywords;
}

@supports (interpolate-size: allow-keywords) {
  details {
    overflow: hidden;
  }

  details::details-content {
    block-size: 0;
      transition: 
        block-size 170ms,
        content-visibility 170ms;
      transition-behavior: allow-discrete;
  }

  details[open]::details-content {
    block-size: auto;
  }
}
